Buffer Events

The following binds to buffer events and logs progress. All buffer events are listed here: 
vWidget.addReadyCallback(function(playerId){
	var vdp = $('#' + playerId )[0];
	}
	vdp.vBind( 'bytesDownloadedChange', function( eventData ){
		$('#bufferEvents').append(
			'testBytesDownloadedChange: ' + eventData.newValue 
		)
	});
	vdp.vBind( 'bufferProgress',  function( eventData ){
		$('#bufferEvents').append(
			'bufferProgress: new time: ' + eventData.newTime	
		)
	});
})
