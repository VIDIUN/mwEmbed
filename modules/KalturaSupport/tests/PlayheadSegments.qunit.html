Playhead Segment support


Sample code for switching segments: 
vWidget.addReadyCallback(function( playerId ){
	var vdp = $('#' + playerId)[0];
	$('#switchTime').click(function () {
		vdp.setVDPAttribute('mediaProxy', 'mediaPlayFrom', $('#v-from').val());
		vdp.setVDPAttribute('mediaProxy', 'mediaPlayTo', $('#v-to').val());
		setTimeout(function(){
			vdp.sendNotification('doPlay');
		},500);
		return false;
	});
});
Switch segment to to GO

This example highlights change the entry to 1_sf5ovm7u and update time 5 to 30 second, followed by a content play
changeMedia 
vWidget.addReadyCallback(function (playerId) {
	var vdp = $('#' + playerId)[0];
	$('#changeMediaAndTime').click(function(){
		vdp.vBind('onChangeMediaDone', function(){
			vdp.sendNotification('doPlay');
		});
		vdp.sendNotification('changeMedia', { 'entryId': '1_sf5ovm7u'} );
		vdp.setVDPAttribute('mediaProxy', 'mediaPlayFrom', 5);
		vdp.setVDPAttribute('mediaProxy', 'mediaPlayTo', 30);
		return false;
	});
});
