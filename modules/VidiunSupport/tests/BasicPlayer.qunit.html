Object Rewrite


Basic player rewrites work by including the javacript library followed by the flash object tag embed:
Note this rewrite method is deprecated. You should instead use vWidget embed. 

<object id="vidiun_player" name="vidiun_player" type="application/x-shockwave-flash" allowFullScreen="true" 
allowNetworking="all" allowScriptAccess="always" height="333" width="400" xmlns:dc="http://purl.org/dc/terms/" 
xmlns:media="http://search.yahoo.com/searchmonkey/media/" rel="media:video" 
resource="http://www.vidiun.com/index.php/vwidget/cache_st/1292436446/wid/_243342/uiconf_id/12905712/entry_id/0_uka1msg4" 
data="http://www.vidiun.com/index.php/vwidget/cache_st/1292436446/wid/_243342/uiconf_id/12905712/entry_id/0_uka1msg4">
	<param name="allowFullScreen" value="true" />
	<param name="allowNetworking" value="all" />
	<param name="allowScriptAccess" value="always" />
	<param name="bgcolor" value="#000000" />
	<param name="flashVars" value="&externalInterfaceDisabled=false" />
	<param name="movie" value="http://www.vidiun.com/index.php/vwidget/cache_st/1292436446/wid/_243342/uiconf_id/12905712/entry_id/0_uka1msg4" />
</object>
