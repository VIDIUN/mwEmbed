Example getting VS via service and used in embed code:

// for example vs service API call see: 
// https://github.com/vidiun/mwEmbed/blob/master/modules/VidiunSupport/apiServices/mweApiVSTest.php
$.getJSON( 'YOUR_ENTITLEMENT_CHECK_SERVICE__&entry_id=1_20x0ca3l&wid=_243342', function( vsJSON ){
	if( ! vsJSON || ! vsJSON.vs ){
		$('#playerTarget').text( 'You may not have permission to view this content');
		return ;
	}
	vWidget.embed( 'playerTarget', {
		'wid' : '_243342',
		'uiconf_id' : '8145862',
		'entry_id' : '0_mjbelixh',
		'flashvars':{
			'vs' : vsJSON.vs
		}
	});
});
