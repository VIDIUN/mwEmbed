vWidget Playlist embed

vWidget can also easily be used to embed playlists, simply pass along playlist plugin configuration as flashvars: 
<div id="myEmbedTarget" style="height:365px;width:650px">
<!--  SEO and video metadata go here -->
	<span property="dc:description" content="test folgers coffe"></span>
	<span property="media:title" content="FolgersCoffe.mpeg"></span>
	<span property="media:width" content="400"></span>
	<span property="media:height" content="300"></span>
</div>
<script src="{{HTML5LibraryURL}}"></script>
<script>
	vWidget.embed({
		'targetId': 'myVideoTarget',
		'wid': '_243342',
		'uiconf_id' : '25975211',
		'flashvars': {
			'playlistAPI':{
				'plugin': true,
				'includeInLayout': true,
				'autoPlay': false,
				'autoInsert':true,
				'vpl0Name' : 'test 4 item playlist',
				'vpl0Url' : 'http://www.vidiun.com/index.php/partnerservices2/executeplaylist?uid=&partner_id=243342&subp_id=24334200&format=8&vs={vs}&playlist_id=0_q79jh71b'
			}
		}
	});
</script>
