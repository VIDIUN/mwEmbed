Enable / Disable GUI

Enable GUI | Dissable GUI
Sample API code: 
vWidget.addReadyCallback(function( playerId ){
	var vdp = $('#' + playerId)[0];
	$('#enableGUI').click(function(){
		vdp.sendNotification('enableGui', { 'guiEnabled' : true } );
	});
	$('#disableGUI').click(function(){
		vdp.sendNotification('enableGui', { 'guiEnabled' : false } );
	});
});
