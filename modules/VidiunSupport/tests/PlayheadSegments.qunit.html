Playhead Segment support


Sample code for switching segments: 
<script>
	vWidget.addReadyCallback(function( playerId ){
		var vdp = $('#' + playerId )[0];
		$('#switchTime').click( function(){
			vdp.setVDPAttribute('mediaProxy', 'mediaPlayFrom', $('#v-from').val() );
			vdp.setVDPAttribute('mediaProxy', 'mediaPlayTo', $('#v-to').val() );
			vdp.sendNotification('doSeek', $('#v-from').val() );
			return false;
		});
	});
</script>
