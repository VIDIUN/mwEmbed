Fullscreen on play

This take the player into fullscreen mode with an event binded against doPlay. When you pause the video it restores to window playback.
This page also logs fullscreen events. 
vWidget.addReadyCallback(function( playerId ){
	var vdp = document.getElementById(playerId); 
	vdp.vBind("doPlay", function(){
		vdp.sendNotification('openFullScreen'); 
	}); 
	vdp.vBind("doPause", function(){
		vdp.sendNotification('closeFullScreen');
	});
	vdp.vBind("openFullScreen", function(){
		$("#fslog").append("openFullScreen\n");
	});
	vdp.vBind("closeFullScreen", function(){
		$("#fslog").append("closeFullScreen\n");
	});
});
